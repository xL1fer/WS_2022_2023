BASE <http://zoo.org/>
PREFIX pred: <http://zoo.org/pred/>

#Select all animals that produce milk or eggs
SELECT ?name
WHERE {
    {
	?nurt_s pred:name 'Eggs' .
	?animal_s pred:nurt ?nurt_s .
	?animal_s pred:name ?name
	}
	UNION
	{
	?nurt_s pred:name 'Milk' .
	?animal_s pred:nurt ?nurt_s .
	?animal_s pred:name ?name
    }
}
